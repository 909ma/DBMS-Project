SELECT study_time.id AS 아이디, users.name AS 이름, study_time.subject AS 과목, study_time.study_time AS 시간 FROM study_time JOIN users ON study_time.id = users.id;

CREATE VIEW sstudy AS SELECT study_time.id AS 아이디, users.name AS 이름, study_time.subject AS 과목, study_time.study_time AS 시간 FROM study_time JOIN users ON study_time.id = users.id;

SELECT * FROM sstudy;
SELECT 이름, 과목, ROUND(AVG(시간),1) AS "평균 시간"
FROM sstudy 
--WHERE 이름 = 'madison'
GROUP BY 이름, 과목
ORDER BY 이름;
